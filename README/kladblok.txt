# stop alles wat op 3001 luistert
PORT=3001
PIDS=$(lsof -tiTCP:$PORT -sTCP:LISTEN)
if [ -n "$PIDS" ]; then
  kill -TERM $PIDS
  sleep 2
  PIDS=$(lsof -tiTCP:$PORT -sTCP:LISTEN)
  if [ -n "$PIDS" ]; then
    kill -KILL $PIDS
  fi
fi
# npm "refresh"
rm -rf node_modules package-lock.json
npm install
# start opnieuw
npm run dev
